mv - move file
rm, rmdir - delete file, folder
make C file - vim (file name)
make C exe file - gcc -o (exe file name) (file name)
exe C file - ./(exe file name)

cat - read, write file _ cat > file - make file and write file

git add . _ git commit _ git push _ git pull

mv - move file to dir (mv file dir) or change file name (mv file file1)
rm - delet file _ rm -r - delet dir _ rmdir - delet dir
chmod - change mode (rwx rwx rwx _ 7 7 7)
